CKEditor 5 Hello world

Posted on 2019-06-04 Edited on 2023-11-24 In CKEditor Views: Disqus:

 

CKEditor 5 現有五種 builds

一、Classic editor

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width">
    <script src="https://cdn.ckeditor.com/ckeditor5/12.2.0/classic/ckeditor.js"></script>
    <title>CKEditor 5</title>
</head>
<body>
    <textarea id="editor1"></textarea>
    <script>
        ClassicEditor
            .create(document.querySelector('#editor1'))
            .catch(error => {
                console.error(error);
            });
    </script>
</body>
</html>

五、Document editor

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width">
    <script src="https://cdn.ckeditor.com/ckeditor5/12.2.0/decoupled-document/ckeditor.js"></script>
    <title>CKEditor 5</title>
</head>
<body>
    <div id="toolbar-container"></div>
    <div id="editor1">test</div>
    <script>
        DecoupledEditor
            .create(document.querySelector('#editor1'))
            .then(editor => {
                const toolbarContainer = document.querySelector('#toolbar-container');
                toolbarContainer.appendChild(editor.ui.view.toolbar.element);
            })
            .catch(error => {
                console.error(error);
            });
    </script>
</body>
</html>
